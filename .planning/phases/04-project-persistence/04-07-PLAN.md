**Step 1.1 — Replace `openClmcFromPath` body with the instrumented version below.** The structural changes are: (1) add `console.log` at each branch with the prefix `[useProject] openClmc:`, (2) wrap the `hashPdf` call in its own try/catch and treat any throw as `missing-pdf` (since a hash failure on a path that `existsSync` claimed to find means the file vanished or is locked between the existsSync and stream open — the user-meaningful state is "PDF not usable", which maps to the missing-pdf modal with Browse-to-relink as the recovery), (3) keep the outer try/catch for genuine schema/parse errors only.

```typescript
const openClmcFromPath = useCallback(
  async (clmcPath: string): Promise<ProjectOpenResult> => {
    console.log('[useProject] openClmc: starting', clmcPath)
    try {
      const text = await window.api.readProject(clmcPath)
      const raw = JSON.parse(text) as unknown
      const version =
        raw && typeof raw === 'object' && 'formatVersion' in (raw as Record<string, unknown>)
          ? (raw as { formatVersion: number }).formatVersion
          : 0
      const data = migrate(raw, version)
      console.log('[useProject] openClmc: read OK, formatVersion=', version)

      const resolved = await window.api.resolvePdfPath(
        clmcPath,
        data.pdf.absolutePath,
        data.pdf.relativePath
      )
      if (!resolved) {
        console.log('[useProject] openClmc: resolvePdfPath returned null — emitting missing-pdf')
        return {
          kind: 'missing-pdf',
          expectedPath: data.pdf.absolutePath,
          expectedName: basename(data.pdf.absolutePath),
          data,
          clmcPath
        }
      }
      console.log('[useProject] openClmc: resolved PDF at', resolved.resolvedPath, 'via', resolved.source)

      // Defensive: hashPdf may throw if the file vanishes/locks between existsSync
      // and stream open. Treat that as the same user-visible state as missing-pdf
      // — the recovery (Browse for replacement) is identical.
      let actualHash: string
      try {
        actualHash = await window.api.hashPdf(resolved.resolvedPath)
        console.log('[useProject] openClmc: actualHash=', actualHash.slice(0, 12), '… expected=', data.pdf.sha256.slice(0, 12), '…')
      } catch (hashErr) {
        const msg = hashErr instanceof Error ? hashErr.message : String(hashErr)
        console.log('[useProject] openClmc: hashPdf threw, treating as missing-pdf:', msg)
        return {
          kind: 'missing-pdf',
          expectedPath: data.pdf.absolutePath,
          expectedName: basename(data.pdf.absolutePath),
          data,
          clmcPath
        }
      }

      if (actualHash !== data.pdf.sha256) {
        console.log('[useProject] openClmc: hash mismatch — emitting hash-mismatch')
        return { kind: 'hash-mismatch', resolvedPdfPath: resolved.resolvedPath, data, clmcPath }
      }

      console.log('[useProject] openClmc: hash matches, proceeding to finishOpen')
      return await finishOpen(resolved.resolvedPath, data, clmcPath)
    } catch (err) {
      console.error('[useProject] openClmc failed:', err)
      return { kind: 'error', message: err instanceof Error ? err.message : String(err) }
    }
  },
  [loadPdfFromPath]
)
```

async function finishOpen(
  resolvedPdfPath: string,
  data: ProjectFileV1,
  clmcPath: string
): Promise<ProjectOpenResult> {
  console.log('[useProject] finishOpen: loading PDF from', resolvedPdfPath)
  try {
    const doc = await loadPdfFromPath(resolvedPdfPath)

    if (doc.numPages !== data.pdf.totalPages) {
      console.log('[useProject] finishOpen: page count mismatch — expected', data.pdf.totalPages, 'actual', doc.numPages)
      return {
        kind: 'page-count-mismatch',
        expected: data.pdf.totalPages,
        actual: doc.numPages,
        data,
        clmcPath
      }
    }

    const dims = await perPageDimensions(doc)
    const dimMismatch = data.pages.some((p) => {
      const actual = dims[p.pageIndex]
      if (!actual) return false
      return (
        Math.abs(actual.width - p.dimensions.width) > 0.5 ||
        Math.abs(actual.height - p.dimensions.height) > 0.5
      )
    })
    if (dimMismatch) {
      console.log('[useProject] finishOpen: dimension mismatch — emitting dimension-mismatch')
      return { kind: 'dimension-mismatch', resolvedPdfPath, data, clmcPath }
    }

    hydrateStores(data)
    useProjectStore.getState().setCurrentFilePath(clmcPath)
    useProjectStore.setState({ isDirty: false, lastSavedAt: Date.now() })
    console.log('[useProject] finishOpen: hydrated successfully')
    return { kind: 'ok' }
  } catch (err) {
    console.error('[useProject] finishOpen threw:', err)
    return { kind: 'error', message: err instanceof Error ? err.message : String(err) }
  }
}

  message,
  onDismiss
}: OpenErrorModalProps): React.JSX.Element {
  const okRef = useRef<HTMLButtonElement>(null)

        position: 'fixed', inset: 0, background: 'rgba(0,0,0,0.5)',

  const [pageAbort, setPageAbort] = useState<{ expected: number; actual: number; data: ProjectFileV1; clmcPath: string } | null>(null)
  const [openError, setOpenError] = useState<{ message: string } | null>(null)

  const handleOpenResult = useCallback((r: ProjectOpenResult | null): void => {
    if (!r || r.kind === 'canceled') return
    if (r.kind === 'ok') { setMissing(null); setHashMiss(null); setPageAbort(null); setOpenError(null); return }
    if (r.kind === 'missing-pdf') { setMissing({ data: r.data, clmcPath: r.clmcPath }); return }
    if (r.kind === 'hash-mismatch') { setHashMiss({ resolvedPdfPath: r.resolvedPdfPath, data: r.data, clmcPath: r.clmcPath }); return }
    if (r.kind === 'page-count-mismatch') { setPageAbort({ expected: r.expected, actual: r.actual, data: r.data, clmcPath: r.clmcPath }); return }
    if (r.kind === 'error') { setOpenError({ message: r.message }); return }
    // dimension-mismatch intentionally unhandled in this harness — covered by App.tsx integration
  }, [])

  // Drive the transition once on mount per `result` prop change
  useState(() => { handleOpenResult(result); return null })

  return (
    <>
      {missing && <MissingPdfModal expectedFilename="planA-3p.pdf" originalPath={missing.data.pdf.absolutePath} onBrowse={() => {}} onCancel={() => setMissing(null)} />}
      {hashMiss && <HashMismatchModal onOpenAnyway={() => {}} onCancel={() => setHashMiss(null)} />}
      {pageAbort && <PageCountAbortModal expectedPages={pageAbort.expected} actualPages={pageAbort.actual} onPickAgain={() => {}} onCancel={() => setPageAbort(null)} />}
      {openError && <OpenErrorModal message={openError.message} onDismiss={() => setOpenError(null)} />}
    </>
  )
}
