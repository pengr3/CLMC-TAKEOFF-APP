2. Create `src/renderer/src/hooks/useCalibrationMode.ts`. The hook owns the state machine locally for per-render data (startPoint, endPoint, previewPoint, popupScreenPos, pixelLength) AND writes every `mode` transition into `scaleStore.calibMode` so the Toolbar (and future subscribers) can react without polling. Mirror the pattern of `useViewportControls.ts` (useCallback, stageRef argument).

   ```typescript
   import { useState, useCallback, useEffect, RefObject } from 'react'
   import type Konva from 'konva'
   import type { CalibMode, StagePoint } from '../types/scale'
   import {
     screenToStage,
     stageToScreen,
     pixelLength,
     MIN_CALIBRATION_PIXELS
   } from '../lib/scale-math'
   import { useScaleStore } from '../stores/scaleStore'

   export interface CalibrationState {
     mode: CalibMode
     startPoint: StagePoint | null
     endPoint: StagePoint | null
     previewPoint: StagePoint | null   // follows cursor between click 1 and click 2
     popupScreenPos: { x: number; y: number } | null
     linePixelLength: number            // in stage-space pixels
     lineTooShort: boolean
   }

   export interface UseCalibrationModeReturn {
     state: CalibrationState
     activate: () => void              // enter 'drawing' from 'idle'
     cancel: () => void                // return to 'idle', clear all points
     recordClick: (screenPos: { x: number; y: number }) => void
     updatePreview: (screenPos: { x: number; y: number }) => void
     recomputePopupPos: () => void     // called when zoom/pan changes while popup visible
   }

   const INITIAL_STATE: CalibrationState = {
     mode: 'idle',
     startPoint: null,
     endPoint: null,
     previewPoint: null,
     popupScreenPos: null,
     linePixelLength: 0,
     lineTooShort: false
   }

   export function useCalibrationMode(
     stageRef: RefObject<Konva.Stage | null>
   ): UseCalibrationModeReturn {
     const [state, setState] = useState<CalibrationState>(INITIAL_STATE)
     // Direct getter — avoids re-subscribing this hook to every store change
     const setCalibModeInStore = useScaleStore((s) => s.setCalibMode)

     const activate = useCallback(() => {
       setState({ ...INITIAL_STATE, mode: 'drawing' })
       setCalibModeInStore('drawing')
     }, [setCalibModeInStore])

     const cancel = useCallback(() => {
       setState(INITIAL_STATE)
       setCalibModeInStore('idle')
     }, [setCalibModeInStore])

     const recordClick = useCallback(
       (screenPos: { x: number; y: number }) => {
         const stage = stageRef.current
         if (!stage) return
         const stagePos = screenToStage(
           screenPos,
           stage.x(),
           stage.y(),
           stage.scaleX()
         )

         setState((prev) => {
           if (prev.mode !== 'drawing') return prev

           if (prev.startPoint === null) {
             // First click — record start, stay in 'drawing'
             return {
               ...prev,
               startPoint: stagePos,
               previewPoint: stagePos
             }
           }

           // Second click — record end, transition to 'confirming'
           const lineLen = pixelLength(prev.startPoint, stagePos)
           const screenEnd = stageToScreen(
             stagePos,
             stage.x(),
             stage.y(),
             stage.scaleX()
           )
           // Push mode transition to store (reactive — replaces 100ms polling)
           setCalibModeInStore('confirming')
           return {
             ...prev,
             mode: 'confirming',
             endPoint: stagePos,
             previewPoint: null,
             popupScreenPos: { x: Math.round(screenEnd.x), y: Math.round(screenEnd.y) },
             linePixelLength: lineLen,
             lineTooShort: lineLen < MIN_CALIBRATION_PIXELS
           }
         })
       },
       [stageRef, setCalibModeInStore]
     )

     const updatePreview = useCallback(
       (screenPos: { x: number; y: number }) => {
         const stage = stageRef.current
         if (!stage) return
         const stagePos = screenToStage(
           screenPos,
           stage.x(),
           stage.y(),
           stage.scaleX()
         )
         setState((prev) => {
           if (prev.mode !== 'drawing' || prev.startPoint === null) return prev
           return { ...prev, previewPoint: stagePos }
         })
       },
       [stageRef]
     )

     // When the stage transform changes while the popup is visible, re-project
     // the end point back to screen-space so the popup tracks the endpoint.
     const recomputePopupPos = useCallback(() => {
       const stage = stageRef.current
       if (!stage) return
       setState((prev) => {
         if (prev.mode !== 'confirming' || prev.endPoint === null) return prev
         const screenEnd = stageToScreen(
           prev.endPoint,
           stage.x(),
           stage.y(),
           stage.scaleX()
         )
         return {
           ...prev,
           popupScreenPos: { x: Math.round(screenEnd.x), y: Math.round(screenEnd.y) }
         }
       })
     }, [stageRef])

     // Escape key cancels at any time during calibration
     useEffect(() => {
       const handleKeyDown = (e: KeyboardEvent): void => {
         if (e.key === 'Escape' && state.mode !== 'idle') {
           e.preventDefault()
           setState(INITIAL_STATE)
           setCalibModeInStore('idle')
         }
       }
       window.addEventListener('keydown', handleKeyDown)
       return () => window.removeEventListener('keydown', handleKeyDown)
     }, [state.mode, setCalibModeInStore])

     return { state, activate, cancel, recordClick, updatePreview, recomputePopupPos }
   }
   ```

3. Create `src/tests/calibration-mode.test.ts`. Since the hook depends on a Konva.Stage ref, tests use a minimal mock stage (duck-typed object with x/y/scaleX methods). The suite validates the state-transition contract that the hook implements, without needing React renderers in node env:
   ```typescript
   import { describe, it, expect, beforeEach } from 'vitest'
   import { screenToStage, pixelLength, MIN_CALIBRATION_PIXELS } from '@renderer/lib/scale-math'
   import { useScaleStore } from '@renderer/stores/scaleStore'
   import { DEFAULT_UNIT } from '@renderer/types/scale'

   // The hook's internal transitions are driven by pure functions that we can
   // test independently. This suite validates the state-transition contract
   // that the hook implements, without needing React renderers in node env.

   describe('calibration state machine contract', () => {
     beforeEach(() => {
       useScaleStore.setState({
         pageScales: {},
         globalUnit: DEFAULT_UNIT,
         calibMode: 'idle'
       })
     })

     // A minimal mock of Konva.Stage's x/y/scaleX getters used by the hook.
     function mockStage(stageX: number, stageY: number, scale: number): {
       x: () => number
       y: () => number
       scaleX: () => number
       scaleY: () => number
     } {
       return {
         x: () => stageX,
         y: () => stageY,
         scaleX: () => scale,
         scaleY: () => scale
       }
     }

     it('first click in drawing mode records stage-space start point', () => {
       const stage = mockStage(100, 50, 2)
       const screenClick = { x: 300, y: 250 }
       const stagePoint = screenToStage(screenClick, stage.x(), stage.y(), stage.scaleX())
       // (300-100)/2 = 100, (250-50)/2 = 100
       expect(stagePoint).toEqual({ x: 100, y: 100 })
     })

     it('two clicks 450 stage-pixels apart produce valid calibration', () => {
       const stage = mockStage(0, 0, 1)
       const start = screenToStage({ x: 0, y: 0 }, stage.x(), stage.y(), stage.scaleX())
       const end = screenToStage({ x: 450, y: 0 }, stage.x(), stage.y(), stage.scaleX())
       const len = pixelLength(start, end)
       expect(len).toBe(450)
       expect(len >= MIN_CALIBRATION_PIXELS).toBe(true)
     })

     it('two clicks less than MIN_CALIBRATION_PIXELS apart flag lineTooShort', () => {
       const stage = mockStage(0, 0, 1)
       const start = screenToStage({ x: 0, y: 0 }, stage.x(), stage.y(), stage.scaleX())
       const end = screenToStage({ x: 5, y: 0 }, stage.x(), stage.y(), stage.scaleX())
       const len = pixelLength(start, end)
       expect(len).toBe(5)
       expect(len < MIN_CALIBRATION_PIXELS).toBe(true)
     })

     it('clicks made at different zoom levels produce identical stage-space lines', () => {
       // Pitfall 1: line length must be stage-space invariant under zoom
       const start_zoom1 = screenToStage({ x: 100, y: 100 }, 0, 0, 1)
       const end_zoom1 = screenToStage({ x: 550, y: 100 }, 0, 0, 1)
       const len_zoom1 = pixelLength(start_zoom1, end_zoom1)

       // Same two physical points in the world, but now stage is at scale 2 (zoomed in)
       // The same physical plan points now have screen positions 200/1100
       const start_zoom2 = screenToStage({ x: 200, y: 200 }, 0, 0, 2)
       const end_zoom2 = screenToStage({ x: 1100, y: 200 }, 0, 0, 2)
       const len_zoom2 = pixelLength(start_zoom2, end_zoom2)

       // Stage-space line lengths should be identical (both = 450 stage-space pixels)
       expect(len_zoom1).toBe(450)
       expect(len_zoom2).toBe(450)
       expect(len_zoom1).toBe(len_zoom2)
     })

     it('setCalibMode writes to scaleStore for reactive subscribers (replaces polling)', () => {
       // The hook must write mode transitions into scaleStore.calibMode.
       // Subscribers (Toolbar) can then react without polling.
       const s = useScaleStore.getState()
       expect(s.calibMode).toBe('idle')
       s.setCalibMode('drawing')
       expect(useScaleStore.getState().calibMode).toBe('drawing')
       s.setCalibMode('confirming')
       expect(useScaleStore.getState().calibMode).toBe('confirming')
       s.setCalibMode('idle')
       expect(useScaleStore.getState().calibMode).toBe('idle')
     })
   })
   ```

```typescript
import { useState, useEffect, useRef, useMemo } from 'react'
import { COLORS } from '../lib/constants'
import type { ScaleUnit } from '../types/scale'
import { useScaleStore } from '../stores/scaleStore'
import {
  computePixelsPerMm,
  formatScaleRatio,
  MIN_CALIBRATION_PIXELS
} from '../lib/scale-math'

export interface ScalePopupProps {
  /** Screen-space position where the line ended (relative to canvas container top-left). */
  screenPos: { x: number; y: number }
  /** Dimensions of the canvas container for clamp computation. */
  containerSize: { width: number; height: number }
  /** Stage-space pixel length of the drawn line. */
  pixelLength: number
  /** Called with (pixelsPerMm, displayUnit) when user confirms with valid input. */
  onConfirm: (pixelsPerMm: number, displayUnit: ScaleUnit) => void
  /** Called when user clicks Discard line or presses Escape. */
  onCancel: () => void
}

const UNITS: ScaleUnit[] = ['mm', 'cm', 'm', 'in', 'ft']
const POPUP_MIN_WIDTH = 220
const POPUP_MAX_WIDTH = 260
const POPUP_HEIGHT_ESTIMATE = 180
const POPUP_OFFSET_X = 8
const POPUP_EDGE_PADDING = 8

export function ScalePopup({
  screenPos,
  containerSize,
  pixelLength,
  onConfirm,
  onCancel
}: ScalePopupProps): React.JSX.Element {
  const globalUnit = useScaleStore((s) => s.globalUnit)
  const [distanceStr, setDistanceStr] = useState('')
  const [unit, setUnit] = useState<ScaleUnit>(globalUnit)
  const inputRef = useRef<HTMLInputElement>(null)

  // Autofocus distance input when popup opens (UI-SPEC §Scale Popup fields)
  useEffect(() => {
    inputRef.current?.focus()
  }, [])

  // Parse distance as decimal (locked decision: decimal format only, no fractions)
  const distance = useMemo(() => {
    const trimmed = distanceStr.trim()
    if (trimmed === '') return NaN
    const parsed = Number(trimmed)
    return isFinite(parsed) ? parsed : NaN
  }, [distanceStr])

  const lineTooShort = pixelLength < MIN_CALIBRATION_PIXELS
  const distanceValid = isFinite(distance) && distance > 0
  const canConfirm = !lineTooShort && distanceValid

  // Compute live ratio preview using the unit-agnostic pixelsPerMm ratio
  const ratioText = useMemo(() => {
    if (lineTooShort) return 'Line too short — please draw again.'
    if (distanceStr.trim() === '') return 'Scale: \u2014'
    if (!distanceValid) return 'Enter a distance greater than zero.'
    try {
      const ppm = computePixelsPerMm(pixelLength, distance, unit)
      return `Scale: ${formatScaleRatio(ppm)}`
    } catch {
      return 'Enter a distance greater than zero.'
    }
  }, [lineTooShort, distanceStr, distanceValid, pixelLength, distance, unit])

  const ratioColor = useMemo(() => {
    if (lineTooShort) return COLORS.warning
    if (distanceStr.trim() === '') return COLORS.textSecondary
    if (!distanceValid) return COLORS.warning
    return COLORS.accent
  }, [lineTooShort, distanceStr, distanceValid])

  // Clamp popup position to canvas container bounds (Pitfall 3).
  // NOTE (LOW review concern): Does not clamp to OS window edge — deferred for v1.
  const popupStyle = useMemo(() => {
    const desiredLeft = screenPos.x + POPUP_OFFSET_X
    const desiredTop = screenPos.y
    const maxLeft = Math.max(
      POPUP_EDGE_PADDING,
      containerSize.width - POPUP_MAX_WIDTH - POPUP_EDGE_PADDING
    )
    const maxTop = Math.max(
      POPUP_EDGE_PADDING,
      containerSize.height - POPUP_HEIGHT_ESTIMATE - POPUP_EDGE_PADDING
    )
    const left = Math.min(Math.max(POPUP_EDGE_PADDING, desiredLeft), maxLeft)
    const top = Math.min(Math.max(POPUP_EDGE_PADDING, desiredTop), maxTop)
    return { left, top }
  }, [screenPos, containerSize])

  const handleConfirmClick = (): void => {
    if (!canConfirm) return
    const ppm = computePixelsPerMm(pixelLength, distance, unit)
    onConfirm(ppm, unit)
  }

  const handleKeyDown = (e: React.KeyboardEvent<HTMLDivElement>): void => {
    if (e.key === 'Enter' && canConfirm) {
      e.preventDefault()
      handleConfirmClick()
    }
    // Escape is handled by useCalibrationMode (global listener) — we don't duplicate
  }

  return (
    <div
      role="dialog"
      aria-modal="true"
      aria-label="Set scale"
      onKeyDown={handleKeyDown}
      style={{
        position: 'absolute',
        left: popupStyle.left,
        top: popupStyle.top,
        minWidth: POPUP_MIN_WIDTH,
        maxWidth: POPUP_MAX_WIDTH,
        padding: 16,
        background: COLORS.secondary,
        border: `1px solid ${COLORS.border}`,
        borderRadius: 8,
        boxShadow: '0 4px 12px rgba(0, 0, 0, 0.4)',
        zIndex: 10,
        display: 'flex',
        flexDirection: 'column',
        gap: 8,
        fontSize: 13,
        lineHeight: 1.4,
        color: COLORS.textPrimary
      }}
    >
      {/* Helper hint */}
      <div style={{ fontSize: 13, fontWeight: 400, color: COLORS.textSecondary }}>
        Enter real-world distance
      </div>

      {/* Distance label + input + unit dropdown */}
      <div style={{ display: 'flex', flexDirection: 'column', gap: 4 }}>
        <label
          htmlFor="scale-distance"
          style={{ fontSize: 13, fontWeight: 600, color: COLORS.textPrimary }}
        >
          Distance
        </label>
        <div style={{ display: 'flex', gap: 8 }}>
          <input
            id="scale-distance"
            ref={inputRef}
            type="text"
            inputMode="decimal"
            placeholder="0.00"
            value={distanceStr}
            onChange={(e) => setDistanceStr(e.target.value)}
            aria-describedby="scale-ratio-preview"
            style={{
              flex: 1,
              height: 28,
              padding: '4px 8px',
              background: COLORS.dominant,
              border: `1px solid ${COLORS.border}`,
              borderRadius: 4,
              color: COLORS.textPrimary,
              fontSize: 13,
              fontWeight: 400,
              outline: 'none'
            }}
            onFocus={(e) => {
              e.currentTarget.style.borderColor = COLORS.accent
            }}
            onBlur={(e) => {
              e.currentTarget.style.borderColor = COLORS.border
            }}
          />
          <select
            aria-label="Unit"
            value={unit}
            onChange={(e) => setUnit(e.target.value as ScaleUnit)}
            style={{
              height: 28,
              padding: '4px 8px',
              background: COLORS.dominant,
              border: `1px solid ${COLORS.border}`,
              borderRadius: 4,
              color: COLORS.textPrimary,
              fontSize: 13,
              fontWeight: 400,
              cursor: 'pointer'
            }}
          >
            {UNITS.map((u) => (
              <option key={u} value={u}>
                {u}
              </option>
            ))}
          </select>
        </div>
      </div>

      {/* Live ratio preview / inline error (role=alert for errors so AT announces) */}
      <div
        id="scale-ratio-preview"
        role={lineTooShort || (!distanceValid && distanceStr.trim() !== '') ? 'alert' : undefined}
        style={{
          fontSize: 13,
          fontWeight: 600,
          color: ratioColor
        }}
      >
        {ratioText}
      </div>

      {/* Action row */}
      <div
        style={{
          display: 'flex',
          justifyContent: 'flex-end',
          gap: 8,
          marginTop: 0
        }}
      >
        <button
          type="button"
          onClick={onCancel}
          aria-label="Discard line"
          style={{
            height: 28,
            padding: '4px 8px',
            background: 'transparent',
            border: `1px solid ${COLORS.border}`,
            borderRadius: 4,
            color: COLORS.textPrimary,
            fontSize: 13,
            fontWeight: 600,
            cursor: 'pointer'
          }}
          onMouseEnter={(e) => {
            e.currentTarget.style.background = COLORS.hoverSurface
          }}
          onMouseLeave={(e) => {
            e.currentTarget.style.background = 'transparent'
          }}
          onMouseDown={(e) => {
            e.currentTarget.style.background = COLORS.activeSurface
          }}
          onMouseUp={(e) => {
            e.currentTarget.style.background = COLORS.hoverSurface
          }}
        >
          Discard line
        </button>
        <button
          type="button"
          onClick={handleConfirmClick}
          disabled={!canConfirm}
          aria-disabled={!canConfirm}
          style={{
            height: 28,
            padding: '4px 8px',
            background: COLORS.accent,
            border: 'none',
            borderRadius: 4,
            color: COLORS.textOnAccent,
            fontSize: 13,
            fontWeight: 600,
            cursor: canConfirm ? 'pointer' : 'not-allowed',
            opacity: canConfirm ? 1 : 0.4
          }}
          onMouseEnter={(e) => {
            if (canConfirm) e.currentTarget.style.background = COLORS.accentHover
          }}
          onMouseLeave={(e) => {
            e.currentTarget.style.background = COLORS.accent
          }}
          onMouseDown={(e) => {
            if (canConfirm) e.currentTarget.style.background = COLORS.accentActive
          }}
          onMouseUp={(e) => {
            if (canConfirm) e.currentTarget.style.background = COLORS.accentHover
          }}
        >
          Confirm
        </button>
      </div>
    </div>
  )
}
```
