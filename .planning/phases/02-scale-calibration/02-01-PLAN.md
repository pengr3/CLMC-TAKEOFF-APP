   ```typescript
   import type { StagePoint, ScreenPoint } from '../types/scale'

   /** Minimum pixel length (stage-space) required for a valid calibration line. */
   export const MIN_CALIBRATION_PIXELS = 10

   /** Euclidean pixel length between two stage-space points. */
   export function pixelLength(a: StagePoint, b: StagePoint): number {
     const dx = b.x - a.x
     const dy = b.y - a.y
     return Math.sqrt(dx * dx + dy * dy)
   }

   /**
    * Compute pixelsPerUnit from a drawn line's stage-space pixel length
    * and the user-entered real-world distance.
    *
    * @throws Error 'Line too short' when linePixelLength < MIN_CALIBRATION_PIXELS
    * @throws Error 'Distance must be > 0' when realWorldDistance <= 0
    */
   export function computePixelsPerUnit(
     linePixelLength: number,
     realWorldDistance: number
   ): number {
     if (linePixelLength < MIN_CALIBRATION_PIXELS) {
       throw new Error('Line too short')
     }
     if (realWorldDistance <= 0) {
       throw new Error('Distance must be > 0')
     }
     return linePixelLength / realWorldDistance
   }

   /** Convert a stage-space pixel distance to real-world units using pixelsPerUnit. */
   export function pixelsToRealWorld(pixels: number, pixelsPerUnit: number): number {
     if (pixelsPerUnit <= 0) return 0
     return pixels / pixelsPerUnit
   }

   /** Format as display scale ratio string, e.g. "1:100". Returns "Unknown" for invalid input. */
   export function formatScaleRatio(pixelsPerUnit: number, _unit: string): string {
     if (pixelsPerUnit <= 0 || !isFinite(pixelsPerUnit)) return 'Unknown'
     const ratio = Math.round(pixelsPerUnit)
     return ratio > 0 ? `1:${ratio}` : 'Unknown'
   }

   /**
    * Convert a screen-space point (relative to Stage DOM element top-left)
    * to stage-space (the coordinate system of the PDF image).
    *
    * Formula: stagePoint = (screenPoint - stageOffset) / stageScale
    */
   export function screenToStage(
     screen: ScreenPoint,
     stageX: number,
     stageY: number,
     stageScale: number
   ): StagePoint {
     return {
       x: (screen.x - stageX) / stageScale,
       y: (screen.y - stageY) / stageScale
     }
   }

   /**
    * Convert a stage-space point back to screen-space.
    * Used to position the HTML popup overlay at the projected line endpoint.
    *
    * Formula: screenPoint = stageOffset + stagePoint * stageScale
    */
   export function stageToScreen(
     stage: StagePoint,
     stageX: number,
     stageY: number,
     stageScale: number
   ): ScreenPoint {
     return {
       x: stageX + stage.x * stageScale,
       y: stageY + stage.y * stageScale
     }
   }
   ```

   ```typescript
   import { create } from 'zustand'
   import { ScaleState, DEFAULT_UNIT } from '../types/scale'

   export const useScaleStore = create<ScaleState>((set, get) => ({
     pageScales: {},
     globalUnit: DEFAULT_UNIT,

     setScale: (page, pixelsPerUnit, unit) =>
       set((state) => ({
         pageScales: {
           ...state.pageScales,
           [page]: { pixelsPerUnit, unit }
         },
         globalUnit: unit
       })),

     getScale: (page) => get().pageScales[page] ?? null,

     clearScale: (page) =>
       set((state) => {
         const next = { ...state.pageScales }
         delete next[page]
         return { pageScales: next }
       }),

     resetAll: () => set({ pageScales: {} }),

     setGlobalUnit: (unit) => set({ globalUnit: unit })
   }))
   ```
