   ```typescript
   export type ScaleUnit = 'mm' | 'cm' | 'm' | 'in' | 'ft'

   /**
    * Per-page scale calibration.
    *
    * INTERNAL REPRESENTATION: pixelsPerMm is the canonical, unit-agnostic
    * ratio of stage-space pixels to real-world millimeters. All measurement
    * math multiplies/divides by pixelsPerMm.
    *
    * DISPLAY CONTEXT: displayUnit records the unit the user calibrated with,
    * so the scale ratio and measurements can be formatted back into the
    * user's preferred unit. Changing globalUnit (user preference) does NOT
    * change pixelsPerMm — it only affects how the ratio is displayed.
    */
   export interface PageScale {
     pixelsPerMm: number
     displayUnit: ScaleUnit
   }

   export type CalibMode = 'idle' | 'drawing' | 'confirming' | 'verifying'

   export interface StagePoint {
     x: number
     y: number
   }

   export interface ScreenPoint {
     x: number
     y: number
   }

   export interface ScaleState {
     pageScales: Record<number, PageScale>
     globalUnit: ScaleUnit
     calibMode: CalibMode
     setScale: (page: number, pixelsPerMm: number, displayUnit: ScaleUnit) => void
     getScale: (page: number) => PageScale | null
     clearScale: (page: number) => void
     resetAll: () => void
     setGlobalUnit: (unit: ScaleUnit) => void
     setCalibMode: (mode: CalibMode) => void
   }

   export const DEFAULT_UNIT: ScaleUnit = 'm'

   /**
    * Conversion table: how many millimeters in 1 of each unit.
    * mm is the canonical base unit internally.
    * in and ft use international inch (25.4 mm exactly).
    */
   export const MM_PER_UNIT: Readonly<Record<ScaleUnit, number>> = {
     mm: 1,
     cm: 10,
     m: 1000,
     in: 25.4,
     ft: 304.8
   } as const
   ```

   ```typescript
   import type { StagePoint, ScreenPoint, ScaleUnit } from '../types/scale'
   import { MM_PER_UNIT } from '../types/scale'

   /** Minimum pixel length (stage-space) required for a valid calibration line. */
   export const MIN_CALIBRATION_PIXELS = 10

   /** Euclidean pixel length between two stage-space points. */
   export function pixelLength(a: StagePoint, b: StagePoint): number {
     const dx = b.x - a.x
     const dy = b.y - a.y
     return Math.sqrt(dx * dx + dy * dy)
   }

   /** Convert a value from the given unit to millimeters. */
   export function toMm(value: number, unit: ScaleUnit): number {
     return value * MM_PER_UNIT[unit]
   }

   /** Convert a millimeter value to the given unit. */
   export function fromMm(mm: number, unit: ScaleUnit): number {
     return mm / MM_PER_UNIT[unit]
   }

   /**
    * Compute pixelsPerMm (unit-agnostic canonical ratio) from a drawn line's
    * stage-space pixel length and the user-entered real-world distance + unit.
    *
    * The distance is internally converted to millimeters so the returned
    * ratio is independent of the input unit. A subsequent change to the
    * user's globalUnit preference does NOT invalidate this value.
    *
    * @throws Error 'Line too short' when linePixelLength < MIN_CALIBRATION_PIXELS
    * @throws Error 'Distance must be > 0' when realWorldDistance <= 0
    */
   export function computePixelsPerMm(
     linePixelLength: number,
     realWorldDistance: number,
     unit: ScaleUnit
   ): number {
     if (linePixelLength < MIN_CALIBRATION_PIXELS) {
       throw new Error('Line too short')
     }
     if (realWorldDistance <= 0) {
       throw new Error('Distance must be > 0')
     }
     const mm = toMm(realWorldDistance, unit)
     return linePixelLength / mm
   }

   /**
    * Convert a stage-space pixel distance to real-world units using pixelsPerMm.
    * The result is in the requested output unit (defaults to mm).
    */
   export function pixelsToRealWorld(
     pixels: number,
     pixelsPerMm: number,
     unit: ScaleUnit = 'mm'
   ): number {
     if (pixelsPerMm <= 0) return 0
     const mm = pixels / pixelsPerMm
     return fromMm(mm, unit)
   }

   /**
    * Format as display scale ratio string, e.g. "1:96" or "1:96.3".
    *
    * Preserves integer display when the ratio is effectively whole (|ratio - round(ratio)| < 0.05).
    * Shows one decimal place otherwise to preserve precision on long spans
    * (e.g. a 1:96 drawing does not silently become "1:97" due to over-rounding).
    *
    * Returns "Unknown" for non-positive or non-finite input.
    */
   export function formatScaleRatio(pixelsPerMm: number): string {
     if (!isFinite(pixelsPerMm) || pixelsPerMm <= 0) return 'Unknown'
     // pixelsPerMm is very small (e.g. 0.09) for most real drawings — invert for display ratio.
     // A drawing at "1:N" scale means 1 unit on paper = N units in reality.
     // pixelsPerMm * (mm per 1 drawing unit) = stage pixels per real unit. For display we
     // show the denominator of the ratio approximation.
     const denom = 1 / pixelsPerMm
     if (!isFinite(denom) || denom <= 0) return 'Unknown'
     const rounded1 = Math.round(denom * 10) / 10
     const isWhole = Math.abs(rounded1 - Math.round(rounded1)) < 0.05
     const display = isWhole ? String(Math.round(rounded1)) : rounded1.toFixed(1)
     return `1:${display}`
   }

   /**
    * Convert a screen-space point (relative to Stage DOM element top-left)
    * to stage-space (the coordinate system of the PDF image).
    *
    * Formula: stagePoint = (screenPoint - stageOffset) / stageScale
    */
   export function screenToStage(
     screen: ScreenPoint,
     stageX: number,
     stageY: number,
     stageScale: number
   ): StagePoint {
     return {
       x: (screen.x - stageX) / stageScale,
       y: (screen.y - stageY) / stageScale
     }
   }

   /**
    * Convert a stage-space point back to screen-space.
    * Used to position the HTML popup overlay at the projected line endpoint.
    *
    * Formula: screenPoint = stageOffset + stagePoint * stageScale
    */
   export function stageToScreen(
     stage: StagePoint,
     stageX: number,
     stageY: number,
     stageScale: number
   ): ScreenPoint {
     return {
       x: stageX + stage.x * stageScale,
       y: stageY + stage.y * stageScale
     }
   }
   ```

   ```typescript
   import { create } from 'zustand'
   import { ScaleState, DEFAULT_UNIT } from '../types/scale'

   export const useScaleStore = create<ScaleState>((set, get) => ({
     pageScales: {},
     globalUnit: DEFAULT_UNIT,
     calibMode: 'idle',

     setScale: (page, pixelsPerMm, displayUnit) =>
       set((state) => ({
         pageScales: {
           ...state.pageScales,
           [page]: { pixelsPerMm, displayUnit }
         },
         globalUnit: displayUnit
       })),

     getScale: (page) => get().pageScales[page] ?? null,

     clearScale: (page) =>
       set((state) => {
         const next = { ...state.pageScales }
         delete next[page]
         return { pageScales: next }
       }),

     resetAll: () => set({ pageScales: {} }),

     /**
      * Update the user's unit preference for future calibration input and
      * display formatting. Does NOT modify any stored pixelsPerMm — those are
      * in canonical millimeters and remain valid regardless of display unit.
      */
     setGlobalUnit: (unit) => set({ globalUnit: unit }),

     /**
      * Set the current calibration interaction mode. Stored in the store (not
      * local component state) so the Toolbar can subscribe reactively instead
      * of polling via setInterval.
      */
     setCalibMode: (mode) => set({ calibMode: mode })
   }))
   ```
