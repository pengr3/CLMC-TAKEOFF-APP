**HIGH review concern resolved — explicit v1 pdfBytes ordering:**

```typescript
async function openV1Silent(text: string, clmcPath: string): Promise<ProjectOpenResult> {
  const raw = JSON.parse(text)

  // Step 2: Extract absolutePath BEFORE migrate discards it
  const absPath = (raw as { pdf: { absolutePath: string } }).pdf.absolutePath

  // Step 3: Read PDF bytes from v1.pdf.absolutePath
  const pdfArrayBuf = await window.api.readPdfBytes(absPath)
  const pdfBytes = new Uint8Array(pdfArrayBuf)

  // Step 4: Cache bytes in viewerStore BEFORE migrate (HIGH review concern)
  useViewerStore.getState().setPdfBytes(pdfBytes)

  // Step 5: Migrate (drops absolutePath/relativePath; bytes already cached)
  const { data: validated, wasMigrated } = migrate(raw, 1)

  // Step 6: Load PDF for rendering (also re-caches bytes — intentionally redundant)
  await loadPdfFromBytes(pdfBytes, validated.pdf.originalFilename)

  // Step 7: hydrate + mark dirty
  hydrateStores(validated)
  useProjectStore.getState().setCurrentFilePath(clmcPath)
  if (wasMigrated) useProjectStore.setState({ isDirty: true, lastSavedAt: null })
  return { kind: 'ok' }
}
```

This ordering ensures that if saveProject is called immediately after v1 open, `pdfBytes` is non-null in viewerStore.
