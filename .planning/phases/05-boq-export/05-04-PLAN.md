```typescript
import { useCallback } from 'react'
import { aggregateBoq, findUncalibratedMarkupPages } from '../lib/boq-aggregator'
import type { BoqStructure, ExportResult } from '../lib/boq-types'
import { useProjectStore } from '../stores/projectStore'
import { useViewerStore } from '../stores/viewerStore'

// =============================================================================
// Path helpers — D-17 default save path derivation
// =============================================================================

function basenameAny(p: string): string {
  const i = Math.max(p.lastIndexOf('/'), p.lastIndexOf('\\'))
  return i >= 0 ? p.slice(i + 1) : p
}

function dirnameAny(p: string): string {
  const i = Math.max(p.lastIndexOf('/'), p.lastIndexOf('\\'))
  return i >= 0 ? p.slice(0, i) : ''
}

function stripExt(s: string): string {
  const i = s.lastIndexOf('.')
  return i > 0 ? s.slice(0, i) : s
}

/**
 * D-17: Default filename = `{project-basename}-BOQ.xlsx` (or .csv).
 * project-basename = stripExt(basename(currentFilePath)) if saved, else
 * stripExt(pdf.originalFilename). Default folder = directory of currentFilePath
 * if saved, else empty (Electron's last-used save folder).
 */
function deriveDefaultExportPath(
  currentFilePath: string | null,
  pdfOriginalFilename: string,
  format: 'xlsx' | 'csv'
): string {
  const projectBase = currentFilePath
    ? stripExt(basenameAny(currentFilePath))
    : stripExt(pdfOriginalFilename || 'plan')
  const dir = currentFilePath ? dirnameAny(currentFilePath) : ''
  const filename = `${projectBase}-BOQ.${format}`
  return dir ? `${dir}/${filename}` : filename
}

// =============================================================================
// useExport hook
// =============================================================================

interface UseExportApi {
  /**
   * Run the full export flow: aggregate → uncalibrated check → dialog → write.
   *
   * Returns:
   * - `{ kind: 'needs-uncalibrated-confirmation', uncalibratedPages, structure }`
   *   when ≥1 page has non-count markups but no scale (D-06). Caller (App.tsx)
   *   shows the warning modal; on Continue, caller invokes
   *   `applyExportAfterConfirm(structure)` to resume — without re-aggregating.
   * - `{ kind: 'canceled' }` when isExporting/isSaving is already true (D-19
   *   race guard) OR when the user cancels the save dialog.
   * - `{ kind: 'ok', filePath }` on successful write.
   * - `{ kind: 'error', message }` when window.api.writeBoq* returns
   *   `{ ok: false, reason }`.
   */
  exportBoq: () => Promise<ExportResult>
  /**
   * Continuation entry point used by App.tsx after the user clicks Continue
   * in UncalibratedExportWarningModal. Skips the uncalibrated check and goes
   * straight to dialog → write using the structure captured at modal-show time.
   */
  applyExportAfterConfirm: (structure: BoqStructure) => Promise<ExportResult>
}

export function useExport(): UseExportApi {
  /**
   * Internal: open dialog, branch on chosen format, write, return ExportResult.
   * Wraps everything in setExporting(true)/finally setExporting(false).
   * Pre-condition: caller has already verified isExporting / isSaving guards.
   */
  async function dialogAndWrite(structure: BoqStructure): Promise<ExportResult> {
    const setExporting = useProjectStore.getState().setExporting
    setExporting(true)
    try {
      const { currentFilePath } = useProjectStore.getState()
      const { fileName: pdfName } = useViewerStore.getState()
      const defaultPath = deriveDefaultExportPath(
        currentFilePath,
        pdfName ?? 'plan.pdf',
        'xlsx'
      )
      // Format hint: 'xlsx' is the primary per D-16. The user may switch
      // filters in the dialog; the returned `format` reflects their choice.
      const chosen = await window.api.saveExportDialog(defaultPath, 'xlsx')
      if (!chosen) return { kind: 'canceled' }

      const writeResult =
        chosen.format === 'csv'
          ? await window.api.writeBoqCsv(chosen.filePath, structure)
          : await window.api.writeBoqXlsx(chosen.filePath, structure)

      if (!writeResult.ok) {
        return { kind: 'error', message: writeResult.reason }
      }
      return { kind: 'ok', filePath: chosen.filePath }
    } catch (err) {
      const message = err instanceof Error ? err.message : String(err)
      return { kind: 'error', message }
    } finally {
      setExporting(false)
    }
  }

  const exportBoq = useCallback(async (): Promise<ExportResult> => {
    // D-19: race guard — refuse to start if already saving or exporting.
    const { isExporting, isSaving } = useProjectStore.getState()
    if (isExporting || isSaving) return { kind: 'canceled' }

    const structure = aggregateBoq()
    const uncalibrated = findUncalibratedMarkupPages()
    if (uncalibrated.length > 0) {
      return {
        kind: 'needs-uncalibrated-confirmation',
        uncalibratedPages: uncalibrated,
        structure
      }
    }

    return dialogAndWrite(structure)
  }, [])

  const applyExportAfterConfirm = useCallback(
    async (structure: BoqStructure): Promise<ExportResult> => {
      const { isExporting, isSaving } = useProjectStore.getState()
      if (isExporting || isSaving) return { kind: 'canceled' }
      return dialogAndWrite(structure)
    },
    []
  )

  return { exportBoq, applyExportAfterConfirm }
}
```

KEY INVARIANTS the implementation MUST honor:
- `setExporting(true)` is called only INSIDE `dialogAndWrite` — NOT before the aggregator runs. Reason: the aggregator is fast (synchronous over Zustand state) and we don't want to disable the Toolbar Export button during the millisecond aggregation runs. The race guard relies on isSaving (which IS set during a Save) — if a Save is in flight, we abort before any work.
- The `try/finally` block surrounds the entire dialog + write sequence. `setExporting(false)` MUST run on every exit path (success, cancel, error, throw).
- `applyExportAfterConfirm` checks the race guard AGAIN — between modal show and Continue click, the user could trigger a Save (Ctrl+S). Refusing to proceed in that case is correct.
- `exportBoq` returns the uncalibrated-confirmation kind WITHOUT setting isExporting=true; the modal flow leaves the Export button enabled (UX nuance — user might Cancel, in which case nothing should appear "in flight").
