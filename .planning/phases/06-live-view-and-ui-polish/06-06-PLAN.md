// Key pattern: renderPageToCanvas at line 36
async function renderPageToCanvas(pdfDocument: PDFDocumentProxy, pageNumber: number): Promise<...> {
  const page = await pdfDocument.getPage(pageNumber)
  const viewport = page.getViewport({ scale: renderScale })
  const dpr = window.devicePixelRatio || 1
  const canvas = document.createElement('canvas')
  canvas.width = Math.floor(viewport.width * dpr)
  canvas.height = Math.floor(viewport.height * dpr)
  const transform = dpr !== 1 ? [dpr, 0, 0, dpr, 0, 0] : undefined
  const renderTask = page.render({ canvas, viewport, transform })
  await renderTask.promise
}

Markup overlay draw function (RESEARCH §4 lines 596-638):
```typescript
function drawMarkupOverlay(ctx: CanvasRenderingContext2D, markups: Markup[], thumbnailScale: number, pdfBaseScale: number): void {
  const dpr = window.devicePixelRatio || 1
  const scale = (thumbnailScale / pdfBaseScale) * dpr
  for (const m of markups) {
    ctx.fillStyle = m.color; ctx.strokeStyle = m.color; ctx.lineWidth = 1.5 * dpr
    if (m.type === 'count') {
      ctx.beginPath(); ctx.arc(m.point!.x * scale, m.point!.y * scale, 3 * dpr, 0, Math.PI * 2); ctx.fill()
    } else if (m.type === 'linear') {
      ctx.beginPath()
      m.points!.forEach((p, i) => { const [x,y] = [p.x*scale, p.y*scale]; i===0?ctx.moveTo(x,y):ctx.lineTo(x,y) })
      ctx.stroke()
    } else {
      ctx.beginPath()
      m.points!.forEach((p, i) => { const [x,y] = [p.x*scale, p.y*scale]; i===0?ctx.moveTo(x,y):ctx.lineTo(x,y) })
      ctx.closePath(); ctx.stroke()
    }
  }
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| pdfDocument.getPage() → Thumbnail rasterization | PDF.js worker renders to a canvas; uncancelled render tasks leak memory |
| markupStore subscription → thumbnail overlay | Subscription must be cleaned up on unmount to prevent memory leak |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-06-01 | Denial of Service | PDF.js render task not cancelled on unmount | mitigate | cancelled flag + renderTask.cancel() in useEffect cleanup (mirrors usePdfRenderer.ts pattern) |
| T-06-06-02 | Denial of Service | markupStore subscription leaking on Thumbnail unmount | mitigate | subscribeWithSelector returns unsub — called in useEffect cleanup; clearTimeout for debounce timer also in cleanup |
| T-06-06-03 | Denial of Service | Canvas memory from 50+ thumbnail rasters | mitigate | IntersectionObserver only rasterizes visible tiles; tiles outside viewport show skeleton (no canvas allocated) |
</threat_model>
