**src/renderer/src/components/TotalsPanelHeader.tsx:**

```typescript
import React from 'react'
import { useViewerStore } from '../stores/viewerStore'
import { COLORS } from '../lib/constants'
import type { BoqStructure } from '../lib/boq-types'

interface TotalsPanelHeaderProps {
  boq: BoqStructure
}

export function TotalsPanelHeader({ boq }: TotalsPanelHeaderProps): React.JSX.Element {
  const currentPage = useViewerStore((s) => s.currentPage)
  const { metadata } = boq
  const rows = [
    { label: 'Project:', value: metadata.projectName || '—' },
    { label: 'Plan:', value: metadata.planFilename || '—' },
    { label: 'Pages:', value: metadata.totalPages > 0 ? String(metadata.totalPages) : '—' },
    { label: 'Markups:', value: String(metadata.totalMarkups) },
    { label: 'Page:', value: currentPage > 0 ? String(currentPage) : '—' }
  ]
  return (
    <div style={{
      background: COLORS.secondary,
      padding: 16,
      display: 'flex', flexDirection: 'column', gap: 8,
      borderBottom: `1px solid ${COLORS.border}`,
      flexShrink: 0
    }}>
      {rows.map(({ label, value }) => (
        <div key={label} style={{ display: 'flex', gap: 8, alignItems: 'baseline' }}>
          <span style={{ fontSize: 12, fontWeight: 600, color: COLORS.textSecondary, flexShrink: 0 }}>{label}</span>
          <span style={{ fontSize: 13, fontWeight: 400, color: COLORS.textPrimary, overflow: 'hidden', textOverflow: 'ellipsis', whiteSpace: 'nowrap' }}>{value}</span>
        </div>
      ))}
    </div>
  )
}
```
