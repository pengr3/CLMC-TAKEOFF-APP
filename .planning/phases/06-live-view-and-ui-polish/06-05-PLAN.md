Match resolution helper (RESEARCH §3 lines 518-545):
```typescript
function matchMarkupsOnPage(pageMarkups: Markup[], itemName: string, rowType: BoqItem['type']): Markup[] {
  const underlyingType: Markup['type'] =
    rowType === 'perimeter-length' || rowType === 'perimeter-area' ? 'perimeter' : rowType as Markup['type']
  return pageMarkups.filter((m) => m.name === itemName && m.type === underlyingType)
}

function findPagesWithMatches(pageMarkupsAll: Record<number, Markup[]>, itemName: string, rowType: BoqItem['type']): number[] {
  const out: number[] = []
  for (const [pageStr, list] of Object.entries(pageMarkupsAll)) {
    if (matchMarkupsOnPage(list, itemName, rowType).length > 0) out.push(Number(pageStr))
  }
  return out.sort((a, b) => a - b)
}
```

Add to TotalsRow:
```typescript
const [cycleIndex, setCycleIndex] = useState(0)
const setPage = useViewerStore((s) => s.setPage)
const pageMarkupsAll = useMarkupStore((s) => s.pageMarkups)
const currentPage = useViewerStore((s) => s.currentPage)

const pagesWithMatches = useMemo(
  () => findPagesWithMatches(pageMarkupsAll, itemName, item.type),
  [pageMarkupsAll, itemName, item.type]
)

const handleClick = () => {
  if (pagesWithMatches.length === 0) return
  const nextPage = pagesWithMatches[cycleIndex % pagesWithMatches.length]
  setPage(nextPage)
  const matches = matchMarkupsOnPage(pageMarkupsAll[nextPage] ?? [], itemName, item.type)
  props.onTriggerPulse?.(matches, item.color ?? '#cccccc')
  setCycleIndex((i) => (i + 1) % pagesWithMatches.length)
}

const handleMouseLeave = () => {
  setCycleIndex(0)
  props.onHover([])
}
```
